197 def __init__(self): 

198 """Initialize FirstEventsDetector""" 

199 self._compile_patterns()

201 def _compile_patterns(self): 

202 """Compile regex patterns""" 

203 self._compiled: List[Tuple[FirstEventType, re.Pattern]] = [] 

204 for event_type, patterns in self.FIRST_PATTERNS: 

205 for pattern in patterns: 

206 try: 

207 compiled = re.compile(pattern, re.IGNORECASE) 

208 self._compiled.append((event_type, compiled)) 

209 except re.error: 

210 pass

212 def detect(self, text: str) -> List[Tuple[FirstEventType, str, float]]: 

213 """ 

214 Detect first events in text. 

215  

216 Args: 

217 text: The text to analyze 

218  

219 Returns: 

220 List of (FirstEventType, matched_text, confidence) tuples 

221 """ 

222 detected = [] 

223 text_lower = text.lower() 

224 

225 # Check for generic "first" keywords 

226 has_first_keyword = any(kw in text_lower for kw in self.FIRST_KEYWORDS) 

227 

228 # Check specific patterns 

229 for event_type, pattern in self._compiled: 

230 match = pattern.search(text) 

231 if match: 

232 confidence = 0.9 if has_first_keyword else 0.7 

233 detected.append((event_type, match.group(), confidence)) 

234 

235 return detected
